module traffic_light_tb;

reg clk;
reg rst;

wire red;
wire yellow;
wire green;

// DUT Instantiation
traffic_light uut (
    .clk(clk),
    .rst(rst),
    .red(red),
    .yellow(yellow),
    .green(green)
);

// Clock generation (10 ns period)
always #5 clk = ~clk;

initial
begin
    clk = 0;
    rst = 1;

    #20 rst = 0;   // Release reset

    #200 $finish;  // Stop simulation
end

initial
begin
    $monitor("Time=%0t | RED=%b YELLOW=%b GREEN=%b", $time, red, yellow, green);
end

endmodule
